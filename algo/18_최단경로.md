다익스트라 알고리즘 : 가중치가 있는 경로에서 최단 경로를 찾는 알고리즘
- 음의 가중치를 허용하지 않음(허용하면 벨만-포드 알고리즘)

1. 출발지에서 가까운 정점 선택
2. 선택한 정점을 경유해서 다른 정점 비용 갱신

인접행렬 사용 시

```java
import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.util.Arrays;
import java.util.StringTokenizer;

public class Dijkstra_AdjList {
	public static class Node {
		int vertex;
		int weight;
		Node next;
		
		public Node(int vertex, int weight, Node next) {
			this.vertex = vertex;
			this.weight = weight;
			this.next = next;
		}
	}
	public static void main(String[] args) throws IOException {
		BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
		StringTokenizer st = new StringTokenizer(br.readLine(), " ");
		
		int V = Integer.parseInt(st.nextToken());
		int E = Integer.parseInt(st.nextToken());
		
		st = new StringTokenizer(br.readLine(), " ");
		int start = Integer.parseInt(st.nextToken());
		int end = Integer.parseInt(st.nextToken());
		
		Node[] adjList = new Node[V];
		boolean[] visited = new boolean[V];
		int[] minDist = new int[V];
		
		for(int j =0;j<E;j++) {
			st = new StringTokenizer(br.readLine(), " ");
			int from = Integer.parseInt(st.nextToken());
			int to = Integer.parseInt(st.nextToken());
			int weight = Integer.parseInt(st.nextToken());
			adjList[from] = new Node(to,weight,adjList[from]);
		}
		
		final int INF = Integer.MAX_VALUE;
		
		Arrays.fill(minDist, INF);
		
		minDist[start] = 0;
		
		for(int i = 0;i<V;i++) {
			int cur = -1;
			int min = INF;
			
			for(int j = 0;j<V;j++) {
				if(!visited[j] && min > minDist[j]) {
					cur = j;
					min = minDist[j];
				}
			}
			
			if(cur == -1) break;
			
			visited[cur] = true;
			
			for (Node temp = adjList[cur]; temp != null ; temp = temp.next) {
				if(!visited[temp.vertex] && minDist[temp.vertex] > min + temp.weight) {
					minDist[temp.vertex] = min + temp.weight;
				}
			}
		}
		
		System.out.println(Arrays.toString(minDist));
		System.out.println(minDist[end] != INF ? minDist[end] : -1);
	}

}
```

PQ 사용 시

```java
import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.util.Arrays;
import java.util.PriorityQueue;
import java.util.StringTokenizer;

public class Dijkstra_PQ {
	public static class Node {
		int to;
		int weight;
		Node next;
		
		public Node(int to, int weight, Node next) {
			this.to = to;
			this.weight = weight;
			this.next = next;
		}
	}
	
	public static class Vertex implements Comparable<Vertex> { //PQ에 넣을 정보
		int no, dist;

		public Vertex(int no, int dist) {
			super();
			this.no = no;
			this.dist = dist;
		}
		
		@Override
		public int compareTo(Vertex o) {
			return Integer.compare(this.dist, o.dist);
		}
	}
	public static void main(String[] args) throws IOException {
		BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
		StringTokenizer st = new StringTokenizer(br.readLine(), " ");
		
		int V = Integer.parseInt(st.nextToken());
		int E = Integer.parseInt(st.nextToken());
		
		st = new StringTokenizer(br.readLine(), " ");
		int start = Integer.parseInt(st.nextToken());
		int end = Integer.parseInt(st.nextToken());
		
		Node[] adjList = new Node[V];
		int[] minDist = new int[V];
		
		for(int j =0;j<E;j++) {
			st = new StringTokenizer(br.readLine(), " ");
			int from = Integer.parseInt(st.nextToken());
			int to = Integer.parseInt(st.nextToken());
			int weight = Integer.parseInt(st.nextToken());
			adjList[from] = new Node(to,weight,adjList[from]);
		}
		
		PriorityQueue<Vertex> pq = new PriorityQueue<>();
		final int INF = Integer.MAX_VALUE;
		Arrays.fill(minDist, INF);
		
		minDist[start] = 0;
		pq.offer(new Vertex(start,minDist[start]));
		
		while(!pq.isEmpty()) {
			Vertex cur = pq.poll();
			
			if(minDist[cur.no] < cur.dist) continue;
			
//			if(cur.no == end) break;
			
			for (Node temp = adjList[cur.no]; temp != null ; temp = temp.next) {
				if(minDist[temp.to] > cur.dist + temp.weight) { // cur.dist = minDist[cur.no]
					minDist[temp.to] = cur.dist + temp.weight;
					pq.offer(new Vertex(temp.to, minDist[temp.to]));
				}
			}
		}
		
		System.out.println(Arrays.toString(minDist));
//		System.out.println(minDist[end] != INF ? minDist[end] : -1);
	}

}
```

0-1 BFS : 가중치가 0과 1 두개일 때 사용  
Deque을 사용하여 BFS 구현
- 가중치가 0이면 front에 넣음
- 가중치가 1이면 rear에 넣음
- 항상 가중치가 낮은 경로를 먼저 탐색하도록 보장
